\begin{frame}{Booleans}
	\begin{itemize}
		\pause\item A \textbf{boolean} can have one of two values: \textbf{true} or \textbf{false}
		\pause\item Named after George Boole (1815--1864), British mathematician
		\pause\item Type in both C\# and Python: \pyinline{bool}
		\pause\item In Python, we have the keywords \pyinline{True} and \pyinline{False}
		\pause\item In C\#, we have \csinline{true} and \csinline{false}
		\pause\item Could be represented by a single bit in memory...
		\pause\item ... but since memory is addressed in bytes (or words of multiple bytes),
			usually represented as an \lstinline{int} with $0$ meaning \csinline{false}
			and any non-zero (e.g.\ $1$) meaning \csinline{true}
	\end{itemize}
\end{frame}

\begin{frame}[fragile]{Boolean values}
	\begin{itemize}
		\pause\item The \csinline{if} statement takes a boolean value as its condition:
	\end{itemize}
	\begin{lstlisting}
if (x > 10)
{
    Debug.Log(x);
}
	\end{lstlisting}
	\begin{itemize}
		\pause\item Variables can also store boolean values:
	\end{itemize}
	\begin{lstlisting}
bool result = x > 10;   # result now stores True or False
if (result)
{
    Debug.Log(x);
}
	\end{lstlisting}
\end{frame}
