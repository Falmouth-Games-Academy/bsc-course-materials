\begin{lstlisting}
def decreaseRed(picture, amount):
  pixelMatrix = getPixels(picture)
  for pixel in pixelMatrix:
    value = getRed(pixel)
    setRedPixel(pixel, value * amount)
\end{lstlisting}

Note: This source code excerpt will not work in PyGame.
